#include <stdio.h>

int main ()
{
    char c = 'W';
    char str[10]={'w','w','w','w','w','w','w','w','w'};

    unsigned short us = 1;
    unsigned long ul = 155UL;
    float f = 1.456789;


printf("------------------------------------------------------------\n"); // Заголовок
printf("| c |       ac           |   us    |  ul   |       f       |\n"); // таблицы
printf("|---|--------------------|---------|-------|---------------|\n");
printf("| %-1c | %+18s | %-7hu | %-5lu | %-7.6E |\n", c, str, us, ul, f);
printf("------------------------------------------------------------\n"); // Конец
}
